    <h1>Opportunity Dashboard</h1>

    <div id="opportunities"></div>


    <script>

        async function fetchOpportunities() {

            try {

                const response = await fetch('http://localhost:3000/opportunities'); // Change this to your API URL after deployment

                const data = await response.json();


                const opportunitiesDiv = document.getElementById('opportunities');

                opportunitiesDiv.innerHTML = ''; // Clear previous data


                if (data && data.length > 0) {

                    data.forEach(opportunity => {

                        const opportunityElement = document.createElement('div');

                        opportunityElement.classList.add('opportunity');

                        opportunityElement.innerHTML = `

                            <h3>${opportunity.name}</h3>

                            <p><strong>Status:</strong> ${opportunity.status}</p>

                            <p><strong>Client:</strong> ${opportunity.client_name}</p>

                            <p><strong>Description:</strong> ${opportunity.description}</p>

                        `;

                        opportunitiesDiv.appendChild(opportunityElement);

                    });

                } else {

                    opportunitiesDiv.innerHTML = '<p>No opportunities found.</p>';

                }

            } catch (error) {

                console.error('Error fetching data:', error);

            }

        }


        // Fetch opportunities every 5 minutes (300000 milliseconds)

        setInterval(fetchOpportunities, 300000);


        // Fetch immediately when the page loads

        fetchOpportunities();

    </script>

</body>
